# 快速上手

本节将介绍如何在项目中使用 Element。

# 引入NnUi

# 在main.js中写入以下内容

import Vue from 'vue';
import NnUI from 'nn-components';
import 'nn-components/lib/nn-components.css';
import App from './App.vue';

Vue.use(NnUI);

new Vue({
  el: '#app',
  render: h => h(App)
});
